##### 课外知识

手写map方法

```js
function map(arr,callback){
    if(!Array.isArray(arr) || !arr.length || typeof callback !== 'function'){
        return [];
    }else {
        let result = [];
        for(let i=0,len = arr.length;i<len;i++){
            result.push(callback(arr[i],i,arr));
        }
        return result;
    }
}
```



##### 3.延申函数环境生命周期

```js
function hd(){
    let n = 1;
    return function sum(){
        console.log(++n);
    }
}
let a = hd();
a(); //n=2
a(); //n=3
//a申请到一个内存地址，所以d
```



```js
function hd(){
    let n = 1;
    return function sum(){
        let m = 1;
        function show(){
            console.log(++m);
        }
        show();
    }
}
let a = hd();
a(); //m = 1
a(); //m = 1
// 因为show函数没有被外部引用，所以都是申请同一地址
```
