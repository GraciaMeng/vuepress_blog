##### 15.函数与方法中this的不同

```js
let obj = {
    name:'meng',
    show:function(){
        return this.name //这里的this是对obj对象的引用
    }
}
console.log(obj.show()) //meng
```

```js
console.log(this == window) //true
```

```js
let obj = {
    name:'meng',
    show:function(){
        function render() {
            console.log(this) // window
        }
    }		
}
// show是类方法，render是普通函数，指向window
```

16.通过常量改变this的指针

```js
let obj = {
    site:'meng',
    list:['js','css']
    show:function(){
        const self = this;
        return this.list.map(
            function(value)=>{
         		return 	`${self.site}-${value}`
            }
		)
    }		
}
```

```js
let obj = {
    site:'meng',
    list:['js','css']
    show:function(){
        return this.list.map(
            function(value)=>{
         		console.log(this) //this
            },
            this //第二参数,可以传一个this
		)
    }		
}
```

##### 17.箭头函数带来的this变化实例

```js
let obj = {
    site:'meng',
    list:['js','css']
    show:function(){
        return this.list.map(value=>{
            console.log(this) //指向父级作用域的this,上下文
        })
    }		
}
```

##### 18.this构造原理

```js
function User(name){
	this.name = name
}
let lisi = new User("lisi");
console.log(lisi);
let cms = {url:'meng.com'};
User.call(cms); // call第一个参数改变对象里面的this，第二个参数赋值给变量 
```

##### 22.bind使用

```js

	console.log(this.name)
}
let func = show.bind({name:'meng'})
show.bind({name:'meng'})()//立即执行

let a  = function(){}
b = a.bind()
console.log(a===b) //false


//参数传递
```
